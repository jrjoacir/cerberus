rails generate model organizations_users organization:references user:references
rails generate model organizations_products organization:references product:references
rails generate model features product:references name:string{30}
rails generate model roles organizations_products:references name:string{30}
rails generate model features_roles features:references roles:references
rails generate model users_roles users:references roles:references

uniques and model validations:
- organizations_users -> organization_id, user_id
- organizations_products -> organization_id, product_id
- features -> product_id, name
- roles -> organizations_products_id, name
- features_roles -> feature_id, role_id
- users_roles -> user_id, role_id
